③ 아래 코드 붙여넣기
function doPost(e) { try { const params = JSON.parse(e.postData.contents); const imageBase64 = params.imageBase64; const mimeType = params.mimeType || 'image/jpeg'; const authorName = params.authorName || ''; const authorDept = params.authorDept || ''; const blob = Utilities.newBlob( Utilities.base64Decode(imageBase64), mimeType, 'label_temp.jpg' ); const folder = DriveApp.getRootFolder(); const file = folder.createFile(blob); const resource = { title: 'ocr_temp', mimeType: MimeType.GOOGLE_DOCS, parents: [{ id: folder.getId() }] }; const ocrFile = Drive.Files.insert( resource, blob, { ocr: true, ocrLanguage: 'ko' } ); const doc = DocumentApp.openById(ocrFile.id); const text = doc.getBody().getText(); file.setTrashed(true); DriveApp.getFileById(ocrFile.id).setTrashed(true); const parsed = parseLabel(text); const sheet = SpreadsheetApp .getActiveSpreadsheet() .getActiveSheet(); if (sheet.getLastRow() === 0) { sheet.appendRow([ '저장시각','성명','근무처','기관명', '분류번호','품명','규격','운용부서', '설치장소','비고','취득일','고유번호', '단가','내용연수','원본텍스트' ]); sheet.getRange(1,1,1,15) .setBackground('#1a6b3c') .setFontColor('white') .setFontWeight('bold'); } sheet.appendRow([ new Date().toLocaleString('ko-KR'), authorName, authorDept, parsed.org, parsed.code, parsed.name, parsed.spec, parsed.dept, parsed.location, parsed.note, parsed.date, parsed.uid, parsed.price, parsed.life, text.replace(/\n/g,' ').substring(0,200) ]); return ContentService .createTextOutput(JSON.stringify({ ok: true, parsed: parsed, rawText: text })) .setMimeType(ContentService.MimeType.JSON); } catch(err) { return ContentService .createTextOutput(JSON.stringify({ ok: false, error: err.message })) .setMimeType(ContentService.MimeType.JSON); } } function parseLabel(text) { const result = { org:'', code:'', name:'', spec:'', dept:'', location:'', note:'', date:'', uid:'', price:'', life:'' }; const lines = text.split('\n') .map(l => l.trim()).filter(l => l); if (lines.length > 0 && !lines[0].includes(':') && !lines[0].includes('번호')) { result.org = lines[0]; } lines.forEach(line => { if (line.includes('분류번호')) result.code = extract(line,'분류번호'); else if (line.includes('품명')) result.name = extract(line,'품명'); else if (line.includes('규격')) result.spec = extract(line,'규격'); else if (line.includes('운용부서')) result.dept = extract(line,'운용부서'); else if (line.includes('설치장소')) result.location = extract(line,'설치장소'); else if (line.includes('비고')) result.note = extract(line,'비고'); else if (line.includes('취득일')) result.date = extract(line,'취득일'); else if (line.includes('고유번호')) result.uid = extract(line,'고유번호'); else if (line.includes('단가')) result.price = extract(line,'단가'); else if (line.includes('내용연수')) result.life = extract(line,'내용연수'); }); return result; } function extract(line, key) { return line.replace(key,'') .replace(/[:：\s]+/,'').trim(); }
